Prime or not

let number = parseInt(prompt("Enter a number: "));
let isPrime = true;
for (let i = 2; i < number; i++) {
if (number % i == 0) {
isPrime = false;
break;
}
}
if (isPrime) {
document.write(`${number} is a prime number`);
} else {
document.write(`${number} is a not prime number`);
}
