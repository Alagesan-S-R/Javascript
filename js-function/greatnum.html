function great_number(num1,num2,num3,num4) {

if((num1===num2)||(num1===num3)||(num1===num4)||(num2===num3)||(num2===num4)||(num3===num4))
{
document.write("some num is equal; enter valid num");
}
else if((num1>num2)&&(num1>num3)&&(num1>num4))
{
document.write("num1 is greater");
}
else if((num2>num3)&&(num2>num4))
{
document.write("num2 is greater");
}
else if(num3>num4)
{
document.write("num3 is greater");
}
else
{
document.write("num4 is greater");
}
}

great_number(num1,num2,num3,num4);
